187std::string lexer_component::type_name(lex::type::T lexem_type)
188{
189 std::string text;
190 lex::type::T TypeBits = lexem_type;
191 //int a, c, b;
192 for(const auto &T: lexer_component::type_enums)
193 {
194 if(!TypeBits)
195 break;
196
197 if(T.first & TypeBits)
198 {
199 text += T.second;
200 TypeBits &= ~T.first; // Remove the Bit from the "Sem" field.
201 if(TypeBits)
202 text += ',';
203 }
204 }
205 if(text.empty())
206 text += "Null";
207 return text;
208}

210lex::type::T lexer_component::type_enum(const std::string &named_type)
211{
212 tux::string str = named_type;
213 tux::string::word::list list;
214 lex::type::T tmap = 0;
215
216 auto count = str.words(list, false,":");
217
218 if(!count) return lex::type::Null;
219 for(auto S: list)
220 {
221 for(auto const& [k,v] : lexer_component::type_enums)
222 {
223 if(std::strcmp(v, (*S).c_str()) == 0) tmap |= k;
224 }
225 }
226 return tmap;
227}

145std::string vchar::render(const vchar* _blk, int _width)
146{
147
148 color::pair curcolors = _blk->colors();
149 std::string _o = curcolors();
150 const vchar* c = _blk;
151 vchar ch;
152 for(int x =0; x< _width; x++)
153 {
154 ch = *c++;
155 auto check = ch.colors();
156 if(curcolors.bg != check.bg)
157 {
158 curcolors.bg = check.bg;
159 _o += color::render_background(curcolors.bg);
160 }
161 if(curcolors.fg != check.fg)
162 {
163 curcolors.fg = check.fg;
164 _o += color::render(curcolors.fg);
165 }
166 if(ch.d& UTFBITS)
167 {
168 if(ch.d & Frame)
169 _o += cadre()[ch.frame_id()];
170 else
171 if(ch.d & Accent)
172 _o += accent_fr::data[ch.accent_id()];
173 else
174 if(ch.d & UGlyph)
175 _o += glyph::data[ch.icon_id()];
176 }
177 else
178 _o += ch.ascii();
179 }
180 _o += _eol_;
181 return _o;
182}

225std::string vchar::details() const
226{
227
228 tux::string infos;
229 tux::string utf_info{};
230 if(d & UTFBITS)
231 {
232 switch(d & UTFBITS)
233 {
234 case UGlyph:
235 utf_info | "icon index:" | icon_id();
236 break;
237 case Accent:
238 utf_info | "accent (fr) index:" | accent_id();
239 break;
240 case Frame:
241 utf_info | "frame index:" | frame_id();
242 break;
243 default:break;
244 //throw rem::exception() [rem::fatal() << " Memory corruption into screen::vchar data!"];
245 }
246 }
247 else
248 utf_info | ascii();
249 infos | "| foreground color:" | foreground() | color::name(foreground()) | color::reset | "| background color:" | background() | color::name(background()) | color::reset;
250 infos | " char:['" | colors() | utf_info() | color::reset | "']";
251 if(d & Underline) infos | "|Underline";
252 if(d & Stroke) infos | "|Stroke";
253 if(d & Blink) infos |"|Blink";
254
255 return infos();
256}

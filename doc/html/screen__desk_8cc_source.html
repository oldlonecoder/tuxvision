245book::code screen::push_front(widget_base *wb)
246{
247 // Refresh
248 UPDATE_ZORDER
249
250 auto it = _toplevels_.end();
251
252 if(it = query(wb); it != _toplevels_.end())
253 return book::code::exist;
254
255 if(!wb->is_toplevel())
256 {
257 book::error() << book::fn::fun << book::code::cancel << " widget '" << color::yellow << wb->class_name() << "::" << wb->id() << color::reset << "' is not a top-level widget. Operation Cancelled";
258 return book::code::cancel;
259 }
260
261 _toplevels_.push_back(wb);
262 // --- todo: 'bitblit' wb onto __back_buffer.
263 _dirty_area_ = wb->_geometry_;
264 auto local = wb->_geometry_.tolocal();
265 _dirty_area_ = _dirty_area_ & _geometry_.tolocal();
266 if(!_dirty_area_)
267 {
268 book::message() << book::code::oob << wb->id() << "' is not visible, reject screen update";
269 return book::code::rejected;
270 }
271 local += ui::cxy{_dirty_area_.a};
272 local = _dirty_area_ & local;
273 local -= _dirty_area_.a;
274
275 for(int y = 0; y < local.dwh.h; y++)
276 {
277 wb->peek_xy(local.a+ui::cxy{0,y});
278 auto bbit = screen::__back_buffer_->begin() + _dirty_area_.a.x + _dirty_area_.a.y * _geometry_.dwh.w;
279 std::copy(wb->_iterator_, wb->_iterator_ + *local.width<size_t>(), bbit);
280 }
281 wb->_tli_ = --_toplevels_.end(); //expose(wb->_geometry_);
282
283 UPDATE_ZORDER
284 return book::code::accepted;
285}

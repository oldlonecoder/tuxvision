60book::code begin()
61{
62 if(auto c = terminal::query_winch(); !c)
63 book::error() << book::code::failed << " to get the screen geometry - there will be no boudaries checks. ";
64
65
66 tcgetattr(STDIN_FILENO, &saved_st);
67 new_term = saved_st;
68
69 new_term.c_iflag &= ~(IGNBRK | BRKINT | PARMRK | ISTRIP | IGNCR | IXON );
70 new_term.c_oflag &= ~OPOST;
71 new_term.c_cflag |= (CS8);
72 new_term.c_cflag &= ~PARENB;
73 new_term.c_lflag &= ~(ICANON | ECHO | IEXTEN | TOSTOP | ECHONL);
74 new_term.c_cc[VMIN] = 0;
75 new_term.c_cc[VTIME] = 1;
76 new_term.c_cc[VSTART] = 0;
77 new_term.c_cc[VSTOP] = 0;
78 tcsetattr(STDIN_FILENO, TCSAFLUSH, &new_term);
79
80 switch_alternate();
81
82 book::out() << " console set to raw mode...";
83
84 ::signal(SIGWINCH, &terminal::resize_signal);
85 terminal::cursor_off();
86 terminal::start_mouse();
87 return book::code::done;
88}

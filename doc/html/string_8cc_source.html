235bool string::eat(string::word& w)
236{
237 if(w.last == _d_.end()) return false;
238 ++w.last;
239 if(w.last == _d_.end())
240 {
241 --w.last;
242 return false;
243 }
244 return true;

270{
271 std::string outstr{};

274 {
275 outstr += std::format("0x{:02x}", static_cast<int>(ch));

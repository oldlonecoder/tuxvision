66book::code line::input(const std::vector<std::string_view>& StrArray)
67{
68 auto a = arguments.end();
69
70 if(StrArray.empty())
71 return book::code::empty;
72
73 for(auto sv : StrArray)
74 {
75 auto next = query(sv);
76
77 book::debug() << color::cyan << "line::input" << color::white << '(' << color::yellow << sv << color::white << ") :";
78
79 if(next == arguments.end())
80 {
81 // It is not a switch - so must be an argument data for the CurArg/NextArg...
82 if((a != arguments.end()) &&
83 ((*a)->required > (*a)->count) && ((*a)->required > 0))
84 {
85 (*a)->arguments.emplace_back(sv);
86 ++(*a)->count;
87 (*a)->enabled = true;
88 book::debug() << color::yellow << (*a)->name << color::reset << " arg:" << (*a)->count << " '" << sv << '\'';
89 }
90 else
91 {
92 // ... or non-switch arg
93 defaults.arguments.emplace_back(sv);
94 ++defaults.count;
95 defaults.enabled = true;
96 book::debug() << color::yellow << defaults.name << color::reset << " arg:" << defaults.count << " '" << sv << '\'';
97 }
98 }
99 else
100 {
101 if ((a != arguments.end()) && ((*a)->count < (*a)->required))
102 {
103 book::error() << " argument " << color::yellow << (*a)->name << color::reset << " is missing " << color::red4 << (*a)->required - (*a)->count << color::reset << " arguments / " << color::lime << (*a)->required;
104 return book::code::failed;
105 }
106 a = next;
107 (*a)->enabled = true;
108 }
109 }
110 return book::code::ok;
111}

114book::action line::process()
115{
116 auto r = book::action::end;
117 if (!arguments.empty())
118 {
119 for (auto* arg : arguments)
120 {
121 if (arg->enabled)
122 {
123 if (arg->required > arg->count)
124 {
125 arg->enabled = false;
126 throw book::except() << " cmd_switch '" << color::yellow << arg->name
127 << color::reset << " is missing "
128 << color::red4 << arg->required - arg->count
129 << color::reset << " arguments: requires "
130 << color::lime << arg->required;
131
132 // throw did not work? :
133 book::except() << " if this is shown, it means exceptions are deactivated. i suggest to enable it... ";
134 book::out() << usage();
135 return book::action::leave;
136
137 }
138 if (arg->delegate_call_back(*arg) != book::action::continu)
139 return book::action::leave;
140 }
141 }
142 }
143 if(!defaults.arguments.empty())
144 {
145 if(defaults.enabled)
146 {
147 r = defaults.delegate_call_back(defaults);
148 if (r != book::action::continu)
149 return book::action::leave;
150 }
151 }
152 return book::action::continu;
153}

155node::node(tux::object* a_parent_scope, lex_token* a_token, alu* _a): tux::object(a_parent_scope, a_token ? std::string(a_token->text()) : "node")
156{
157 _token_ = a_token;
158
159 //book::debug() << book::fn::fun << "node::node => token details:[" << (t0 ? t0->Details() : "nullptr") << "]"<< book::fn::endl ;
160
161 if (_a != nullptr)
162 {
163 _a_ = _a;
164 mem.acc = 0;
165 mem.ref = 1;
166 return;
167 }
168
169 _a_ = new alu(0.0f);
170 mem.acc = 1;
171 mem.ref = 0;
172
173 if (!_token_)
174 return;
175
176 switch (_token_->prim) {
177 case lex::type::Text:
178 *_a_ = std::string(a_token->text()); // deep copy 'cause _alu_ is this node's alu
179 return;
180 case lex::type::Number:
181 {
182 if(_token_->sem & lex::type::Keyword)
183 {
184 _op_fn_ = node::s_operator_table.find(a_token->m)->second;
185
186 if(!_op_fn_)
187 throw book::exception() [ book::fatal() << _token_->mark() << book::fn::endl << "as no implementation!" ];
188 return;
189 }
190 if (_token_->sem & lex::type::Const)
191 {
192 double d;
193 (tux::string(a_token->text())) >> d;
194 *_a_ = d;
195 book::debug() << book::fn::fun << " acc: " << color::yellow << (*_a_)();
196 }
197 }
198 return;
199 case lex::type::Hex:
200 {
201 uint64_t d;
202 tux::string(a_token->text()).hex(d);
203 *_a_ = d;
204 return;
205 }
206 return;
207 case lex::type::Any:
208 case lex::type::VoidPtr:
209 *_a_ = static_cast<void *>(nullptr);
210 return;
211 default:
212
213 break;
214 }
215
216 _op_fn_ = node::s_operator_table.find(a_token->m)->second;
217 //book::debug() << book::fn::fun << " acc: '" << color::yellow << (*acc)() << color::white << "'";
218}

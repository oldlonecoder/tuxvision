331book &book::out(std::source_location src)
332{
333 book_guard.lock();
334 if(book::sections.empty())
335 {
336 book::init();
337 auto& r = **book::current_section << book{book::type::warning, book::code::empty, src};
338 r << " using uninitialized book storage - Initializing with default: 'globals'";
339 r << "\r\n";
340 book_guard.unlock();
341 return r;
342 }
343
344 if(!book::current_section->contents.empty())
345 {
346 //@todo apply encoded newline : either ascii or html
347 book::current_section->contents.back().text | "\r\n"; // this way we make sure that we set new line for all oses.
348 book_guard.unlock();
349 return book::current_section->contents.back();
350 }
351 auto& r = **book::current_section << book{book::type::output, book::code::empty, src};
352 book_guard.unlock();
353 return r;

512void book::init_header()
513{
514 if(_headercomp_.type)
515 {
516 auto [gh,colors] = book::type_attributes(_type_);
517 text | colors | gh | ' ' | book::to_string(_type_) | color::reset | ' ';
518 }
519
520 if(_headercomp_.file)
521 {
522 auto [gh, colors] = book::function_attributes(book::fn::file);
523 tux::string txt = location.file_name();
524 tux::string::word::list words;
525 if(auto count = txt.words(words, false, "/"); !count) return ;
526 text | color::reset | " in " | colors | gh;
527 if(words.size() > 1)
528 {
529 auto i = words.end();
530 i--; // filename
531 auto filename = *i;
532 i--; // parent dir
533 text | **i;
534 ++i;
535 text | '/' | **i | ' ';
536 }
537 words.clear();
538 }
539 // disable uncontrollable and potentially insane length of function signature given by std::source_location.
540 // instead, using it explicitly into a distinct book input
541 //------------------------------------------------------------------------------------------------------------
542 // if(headercomp.fun)
543 // {
544 // auto [gh, colors] = book::functionattributes(book::fn::fun);
545 // text << colors << gh << ' ' << location.function_name() << color::reset;
546 // }
547 //------------------------------------------------------------------------------------------------------------
548 if(_headercomp_.line)
549 {
550 auto [gh, colors] = book::function_attributes(book::fn::line);
551 text | colors | "line: " | std::format("{}", location.line()) | color::reset;
552 }
553 text | ' ';
554}

563void book::purge(const std::function<void(book &)>& f)
564{
565 std::cout << " Purge book::sections->contents:\n\r";
566 for(auto& sec: book::sections)
567 {
568 std::cout << " Section: [" << sec.id << "]:\n\r--------------------------------------------------------\n\r";
569 for(auto& r : sec.contents)
570 {
571 if (f)
572 f(r);
573 else
574 std::cout << r.text() << "\r\n";
575
576 r.text.clear();
577 }
578 book::sections[0].contents.clear();
579 }
580}

706{

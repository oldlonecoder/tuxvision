184book::code event::get_stdin_event(event &_event_, timeval tv)
185{
186 //book::log() << book::fn::fun;
187
188 fd_set fds;
189 FD_ZERO(&fds); // NOLINT
190 FD_SET(STDIN_FILENO, &fds); // NOLINT
191 select(STDIN_FILENO + 1, &fds, nullptr, nullptr, &tv); // NOLINT
192
193 if(auto e = FD_ISSET(STDIN_FILENO, &fds); !e)
194 {
195 book::out() << "TIMEOUT";
196 _event_.event_type = event::type::noop;
197 return book::code::timeout;
198 }
199 int pksize{0};
200 ioctl(0,FIONREAD,&pksize);
201 if(!pksize)
202 {
203 book::out() << color::reset << "stdin triggered with 0 bytes ( likely terminal resize event )...";
204 return book::code::empty;
205 }
206 //book::out() << " STD INPUT EVENT : " << color::yellow << pksize << color::reset << " bytes";
207 char buffer[100];
208
209 size_t r = ::read(STDIN_FILENO, buffer, 100);
210 buffer[r] = 0;
211 //book::out() << "{" << color::yellow << tuxin::string::bytes(buffer) << color::reset << "}";
212
213 conio_parser cparser;
214 _event_.event_type = cparser.parse(_event_, buffer);
215
216 if(_event_.event_type == event::type::DROP)
217 return book::code::rejected;
218
219 //if(_event_.event_type == event::type::MOUSE)
220 // book::out() << event::mouse_data.to_string() << color::reset;
221
222 else if(_event_.event_type == event::type::key_command)
223 {
224 book::out() << "KEY COMMAND: {" << color::lime << _event_.data.kev.description << color::reset << '}';
225 if(_event_.data.kev.mnemonic == keyevent::ESC)
226 return book::code::cancel;
227 }
228 return book::code::accepted;
229}

267event::type event::conio_parser::parse(event &_ev_, const char *a_seq_)
268{
269 //book::log() << book::fn::fun;
270 _seq_ = a_seq_;
271 //book::debug() << book::fn::fun << "{" << color::yellow << tuxin::string::bytes(_seq_) << color::reset << "}:";
272 it = _seq_.begin();
273
274 // L'on s'occupe des CSI (Control Sequence Introducer) avant tout:
275 if((*it==27) && (_seq_.size()>1)) return parse_esc(_ev_);
276 if(*it == 27)
277 {
278 // systematic/explicitely ESC pressed:
279 _ev_.data.kev = keyevent::query(27l);
280 book::out() << _ev_.data.kev.description ;
281 _ev_.event_type = event::type::key_command;
282 return _ev_.event_type;
283 }
284
285 //---------------------------------------------------------------
286 // Not ESC seq.
287 // Check CTRL hit ( *it < ascii(32) ) + ENTER,TAB, BACKSP ...
288 // One Byte key. All extra keys are ignored/discarded.
289
290 if((*it < 32) || (*it == 127))
291 {
292
293 _ev_.event_type = event::type::key_command;
294 book::out() << "KEY_COMMAND: " << color::yellow << (int)*it << color::reset << "';";
295 if(keyevent tmpkev = keyevent::query(*it); tmpkev.mnemonic != keyevent::None)
296 {
297 _ev_.data.kev=tmpkev;
298 return _ev_.event_type;
299 }
300 _ev_.data.kev.code = *it;
301 return _ev_.event_type;
302 }
303 // -- key input.
304 book::out() << "CHARACTER:'" << color::yellow << *it << color::reset << "';";
305 _ev_.data.kev = keyevent::nokey;
306 _ev_.event_type = event::type::CHARACTER; // Extra bytes are all ignored.
307 _ev_.data.kev.code = *it;
308 return _ev_.event_type;
309}

312event::type event::conio_parser::parse_esc(event& evd)
313{
314 //book::log() << book::fn::fun;
315
316 if (!next_byte()) return event::type::UNCOMPLETED;
317 switch (*it)
318 {
319 case 'P':
320 return parse_dcs();
321 case '[':
322 return parse_csi(evd);
323 case ']':
324 return parse_osc(evd);
325
326 // Expecting 2 characters.
327 case ' ':
328 case '#':
329 case '%':
330 case '(':
331 case ')':
332 case '*':
333 case '+':
334 case 'O':
335 case 'N': {
336 if (!next_byte()) return event::type::UNCOMPLETED;
337 return parse_ss_1_2(evd);
338 }
339 // Expecting 1 character:
340 default:
341 return event::type::SPECIAL;
342 }

345event::type event::conio_parser::parse_ss_1_2(event &evd)
346{
347 // 8 bytes max which include the beginning of the buffer (ESC;O | [)
348
349 // Consume the buffer. keyevent::ansi_seq is now the final location.
350 //book::debug() << book::fn::fun;
351 //book::out() << "copy seq into event kev ansi_seq: ";
352 std::strncpy(evd.data.kev.ansi_seq, _seq_.data(), 19);
353
354 auto& code = evd.data.kev.code;
355 code = 0;
356 auto ln = std::min(size_t(8),std::strlen(evd.data.kev.ansi_seq));
357 //book::out() << " Check: ansi_seq to parse: {" << color::yellow << tuxin::string::bytes(evd.data.kev.ansi_seq) << color::reset << "}:";
358
359 U8 _b_ = evd.data.kev.ansi_seq[0]; // start with ESC byte
360 code = (code << 8) | _b_;
361 _b_ = evd.data.kev.ansi_seq[1]; // no matter '[' or 'O' :
362 code = (code << 8) | _b_;
363 // Now push the rest towards ln or until hit or miss:
364 for(int x=2;x < ln; x++)
365 {
366 _b_ = evd.data.kev.ansi_seq[x];
367 code = (code << 8) | _b_;
368 book::out() << std::format("{:04x}",code);
369 if(keyevent tmp_kev = keyevent::query(code); tmp_kev.mnemonic != keyevent::None)
370 {
371 book::out() << "Match:" << color::lime << tmp_kev.description << color::reset;
372 evd.data.kev = tmp_kev;
373 evd.event_type = event::type::key_command;
374 return evd.event_type;
375 }
376 }
377
378 return event::type::noop;
379}

387event::type event::conio_parser::parse_csi(event& evd)
388{
389
390 //book::debug() << book::fn::fun;
391
392 bool altered = false;
393 int argument = 0;
394 std::vector<int> arguments;
395 while (true) {
396 if (!next_byte()) {
397 //book::out() << " End of the sequence ";
398 return event::type::UNCOMPLETED;
399 }
400
401 if (*it == '<') {
402 //book::out() << " altered mode";
403 altered = true; // Just bypass ...for now...
404 continue;
405 }
406
407 if (*it >= '0' && *it <= '9') {
408 //book::out() << "cursor on digit: '" << color::yellow << (*it-'0') << color::reset << "'";
409 argument *= 10; // NOLINT
410 argument += *it - '0';
411 continue;
412 }
413
414 if (*it == ';') {
415 //book::out() << "arg separator: arg value: '" << color::yellow << argument << color::reset << "' ";
416 arguments.push_back(argument);
417 argument = 0;
418 continue;
419 }
420
421 // CSI is terminated by a character in the range 0x40–0x7E
422 // (ASCII @A–Z[\]^_`a–z{|}~),
423 // Note: I don't remember why we exclude '<'
424 // To handle F1-F4, we exclude '['.
425 if (*it >= '@' && *it <= '~' && *it != '<' && *it != '[')
426 {
427 //book::out() << "CSI completed: last arg : '" << color::yellow << argument << color::reset << "' ";
428 arguments.push_back(argument);
429 argument = 0; // NOLINT
430 int c=1;
431 //for(auto n : arguments) book::out() << c++ << " > " << std::format("0x{:02x}",n);
432 switch (*it) {
433 case 'M':
434 return parse_mouse(evd,altered, true, std::move(arguments));
435 case 'm':
436 return parse_mouse(evd,altered, false, std::move(arguments));
437 case 'R':
438 return parse_caret_report(std::move(arguments));
439 default:
440 book::out() << " Switching to parse_ss_1_2():";
441 return parse_ss_1_2(evd);
442 }
443 }
444 book::out() << " Unterminated CSI: switchting to parse_ss_1_2";
445 return parse_ss_1_2(evd);
446
447 // Invalid ESC in CSI.
448 if (*it == '\x1B') {
449 return event::type::SPECIAL;
450 }
451 }
452}

542event::type event::conio_parser::parse_utf8(event& ev)
543{
544 auto head = *it;
545 unsigned char selector = 0b1000'0000; // NOLINT
546
547 // The non code-point part of the first byte.
548 unsigned char mask = selector;
549
550 // Find the first zero in the first byte.
551 unsigned int first_zero = 8; // NOLINT
552 for (unsigned int i = 0; i < 8; ++i) { // NOLINT
553 mask |= selector;
554 if (!(head & selector)) {
555 first_zero = i;
556 break;
557 }
558 selector >>= 1U;
559 }
560
561 // Accumulate the value of the first byte.
562 auto value = uint32_t(head & ~mask); // NOLINT
563
564 // Invalid UTF8, with more than 5 bytes.
565 const unsigned int max_utf8_bytes = 5;
566 if (first_zero == 1 || first_zero >= max_utf8_bytes) {
567 ev.event_type = event::type::DROP;
568 return event::type::DROP;
569 }
570
571 // Multi byte UTF-8.
572 for (unsigned int i = 2; i <= first_zero; ++i) {
573 if (!next_byte()) {
574 ev.event_type = event::type::UNCOMPLETED;
575 return event::type::UNCOMPLETED;
576 }
577
578 // Invalid continuation byte.
579 head = *it;
580 if ((head & 0b1100'0000) != 0b1000'0000) { // NOLINT
581 ev.event_type = event::type::DROP;
582 return event::type::DROP;
583 }
584 value <<= 6; // NOLINT
585 value += head & 0b0011'1111; // NOLINT
586 }
587
588 // Check for overlong UTF8 encoding.
589 int extra_byte = 0;
590 if (value <= 0b000'0000'0111'1111) { // NOLINT
591 extra_byte = 0; // NOLINT
592 } else if (value <= 0b000'0111'1111'1111) { // NOLINT
593 extra_byte = 1; // NOLINT
594 } else if (value <= 0b1111'1111'1111'1111) { // NOLINT
595 extra_byte = 2; // NOLINT
596 } else if (value <= 0b1'0000'1111'1111'1111'1111) { // NOLINT
597 extra_byte = 3; // NOLINT
598 } else {
599 ev.event_type = event::type::DROP; // NOLINT
600 return event::type::DROP;
601 }
602
603 if (extra_byte != (it - _seq_.begin())) {
604 ev.event_type = event::type::DROP;
605 return event::type::DROP;
606 }
607 ev.event_type = event::type::CHARACTER;
608 return event::type::CHARACTER;
609}

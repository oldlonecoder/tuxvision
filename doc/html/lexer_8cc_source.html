227book::code lexer::tokenize_sign_prefix(lex_token &newtoken)
228{
229 book::debug() << book::fn::fun << ":";
230 if ((newtoken.m != lex::mnemonic::Sub) && (newtoken.m != lex::mnemonic::Add))
231 {
232 book::out() << " rejected (not subtract or addition operator).";
233 return book::code::rejected;
234 }
235
236 if(!m_config.production->empty())
237 {
238 auto &token = m_config.production->production_table().back();
239 book::out() << "prev: " << token.details(true);
240 if((!token.has_type(lex::type::Binary |lex::type::Punc)) || (token.m == lex::mnemonic::ClosePar))
241 {
242 book::out() << " rejected: previous token is not an operator or a expression-terminating punctuation token";
243 return book::code::rejected;
244 }
245 }
246
247 book::out() << " conditions met for unary prefix sign operator:";
248 newtoken.prim =lex::type::Sign;
249 newtoken.sem = (newtoken.sem & ~(lex::type::Postfix|lex::type::Binary)) | lex::type::Prefix |lex::type::Unary | lex::type::Operator| lex::type::Sign;
250 newtoken.m = lex::mnemonic::Negative;
251 newtoken.token_location.length = 1;
252 newtoken.token_location.end = newtoken.token_location.begin + 1;
253 return book::code::accepted;

265book::code lexer::tokenize_prefix(lex_token &newtoken)
266{
267 if ((newtoken.m == lex::mnemonic::Decr) || (newtoken.m == lex::mnemonic::Incr) || (newtoken.m == lex::mnemonic::BinaryNot))
268 {
269 if(production().empty())
270 {
271 push_token(newtoken);
272 return book::code::accepted;
273 }
274 if((production().back().flags.V) && (!production().back().has_type(lex::type::Operator)))
275 {
276 newtoken.prim = lex::type::Postfix;
277 newtoken.sem &= ~lex::type::Prefix |lex::type::Postfix;
278 if(newtoken.m == lex::mnemonic::BinaryNot)
279 newtoken.m = lex::mnemonic::Factorial;
280 }
281 if(!production().back().flags.V)
282 throw book::syntax() << newtoken.details() << ": illegal;";
283 }
284 push_token(newtoken);
285
286 return book::code::accepted;
287}
